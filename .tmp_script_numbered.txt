  224:   const parser = new DOMParser();

  228:   function walk(node, path = []) {
  229:     const tag = (node.tagName || '').toUpperCase();
  230: 
  231:     if (tag === 'A') {
  232:       const title = (node.textContent || '').trim();
  233:       const url = (node.getAttribute('href') || '').trim();
  234:       if (title && /^https?:\/\//i.test(url)) {
  235:         let favicon = '';
  236:         try {
  237:           favicon = `https://www.google.com/s2/favicons?domain=${new URL(url).hostname}&sz=16`;
  238:         } catch (_) {
  239:           favicon = '';
  240:         }
  241:         const category = path.length > 1 ? path[1] : 'Œ¥∑÷¿‡';
  242:         parsedBookmarks.push({
  243:           id: generateId(),
  244:           title,
  245:           url,
  246:           category,
  247:           favicon,
  248:           originalPath: path.join('/'),
  249:           createdAt: new Date().toISOString(),
  250:           updatedAt: new Date().toISOString()
  251:         });
  252:       }
  253:       return;
  254:     }
  255:

  257:       const folderName = (node.textContent || '').trim();
  258:       let dl = node.nextElementSibling;
  259:       while (dl && (dl.tagName || '').toUpperCase() !== 'DL') {
  260:         dl = dl.nextElementSibling;

  264:       }

  266:     }
  267: 
  268:     Array.from(node.children || []).forEach((child) => walk(child, path));
  269:   }

  629: async function checkUrlWithRetries(url, retries, timeoutMs) {
  630:   const attempts = [];
  631:   let totalTime = 0;
  632:   let lastReason = 'Œ¥÷™¥ÌŒÛ';
  633: 
  634:   for (let i = 1; i <= retries; i += 1) {
  635:     const start = Date.now();
  636:     try {
  637:       const ok = await checkUrlReachable(url, timeoutMs);
  638:       const duration = Date.now() - start;
  639:       totalTime += duration;
  640:       attempts.push({ attemptNo: i, ok, durationMs: duration, reason: ok ? '' : '«Î«Û ß∞‹' });
  641:       if (ok) {
  642:         return { ok: true, attempts, latencyMs: Math.round(totalTime / attempts.length), reason: '' };
  643:       }
  644:       lastReason = '«Î«Û ß∞‹';
  645:     } catch (err) {
  646:       const duration = Date.now() - start;
  647:       totalTime += duration;
  648:       const reason = err && err.message ? err.message : 'Õ¯¬Á¥ÌŒÛ';
  649:       attempts.push({ attemptNo: i, ok: false, durationMs: duration, reason });
  650:       lastReason = reason;
  651:     }
  652:   }
  653: 
  654:   return { ok: false, attempts, latencyMs: Math.round(totalTime / attempts.length), reason: lastReason };
  655: }

  657: async function checkUrlReachable(url, timeoutMs) {
  658:   const controller = new AbortController();
  659:   const timer = setTimeout(() => controller.abort(), timeoutMs);
  660: 
  661:   try {
  662:     const headRes = await fetch(url, {

  665:       redirect: 'follow',
  666:       signal: controller.signal,
  667:       cache: 'no-store'
  668:     });
  669:     if (headRes.type === 'opaque' || headRes.ok) return true;
  670:   } catch (_) {
  671:     // HEAD  ß∞‹∫Û≥¢ ‘ GET

  679:       signal: controller.signal,
  680:       cache: 'no-store'
  681:     });
